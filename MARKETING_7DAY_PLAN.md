### ✅ Exit Intent 팝업 (이탈 방지)

```javascript
// public/static/exit-intent.js
let exitPopupShown = false;

document.addEventListener('mouseleave', (e) => {
  // 마우스가 화면 상단을 벗어날 때 (페이지 닫기 시도)
  if (e.clientY < 0 && !exitPopupShown && !localStorage.getItem('exit_popup_today')) {
    showExitPopup();
    exitPopupShown = true;
  }
});

function showExitPopup() {
  const popup = document.createElement('div');
  popup.className = 'exit-popup';
  popup.innerHTML = `
    <div class="exit-popup-overlay">
      <div class="exit-popup-content">
        <button class="close-btn" onclick="closeExitPopup()">×</button>
        
        <div class="icon">🎁</div>
        <h2>잠깐만요!</h2>
        <p>지금 가입하시면 <strong>첫 50명 한정</strong></p>
        <p class="highlight">200 포인트 무료 증정!</p>
        <p class="small">(₩200 상당, AI 멘토 2회 사용 가능)</p>
        
        <button class="cta-btn" onclick="window.location.href='/signup.html?promo=EXIT200'">
          무료 포인트 받기
        </button>
        
        <p class="timer">⏰ 이 혜택은 <span id="countdown">60</span>초 후 사라집니다</p>
      </div>
    </div>
  `;
  document.body.appendChild(popup);
  
  // 60초 카운트다운
  let seconds = 60;
  const timer = setInterval(() => {
    seconds--;
    document.getElementById('countdown').textContent = seconds;
    if (seconds <= 0) {
      clearInterval(timer);
      closeExitPopup();
    }
  }, 1000);
  
  // 오늘 하루 다시 보지 않기
  localStorage.setItem('exit_popup_today', new Date().toDateString());
}

function closeExitPopup() {
  document.querySelector('.exit-popup')?.remove();
}
```

**예상 효과:**
- 이탈률 -20% (기존 40% → 개선 32%)
- Exit Popup 전환율 5-8%
- 추가 가입자: 10-15명/일
